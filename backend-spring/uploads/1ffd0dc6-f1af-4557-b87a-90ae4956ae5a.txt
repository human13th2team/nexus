create table user(
user_id UUID not null auto_increment,  -- 고유 번호
nickname varchar(255),  -- 닉네임
email varchar(255),  -- 이메일(아이디)
passwd varchar(255), -- 비밀번호(암호화)
user_type int default 0, -- 0 일반회원 1 사업가 회원
biz_no varchar(12), -- 사업가 번호
adress varchar(255), -- 주소
login_ty int default 0, -- 0 로컬 1 구글 2 카카오
access_token varchar(255), -- api연동
day timestamp default now(), -- 등록일
primary key(user_id)
)

create table board (
    board_id UUID not null auto_increment,  -- 게시판 고유 번호
    user_id UUID not null,  -- 유저 고유 번호
    board_title varchar(100) not null, -- 게시판 제목
    board_content text,  -- 게시판 내용

    region_name varchar(20),  -- 지역
    category_name varchar(20),       -- 업종
    
    view_count int default 0, -- 조회수
    board_day timestamp default now(),  -- 게시판 등록일
    board_img varchar(255),  -- 게시판 이미지 업로드
    is_anonymous tinyint(1) default 0, -- 익명
    
    primary key(board_id),
    index idx_list (region_name, board_day desc),
    foreign key(user_id) references user(user_id) on delete cascade
)


CREATE TABLE comments (
    comment_id UUID AUTO_INCREMENT PRIMARY KEY,
    board_id UUID NOT NULL,                -- 게시판 번호
    user_id UUID NOT NULL,                 -- 유저 번호
    comment_content TEXT NOT NULL,        -- 댓글/대댓글 통합 내용
    parent_comment_id INT DEFAULT NULL,    -- 대댓글 핵심: 자기 자신의 comment_id를 참조함
    comment_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, -- 댓글 등록일
    FOREIGN KEY (board_id) REFERENCES board(board_id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES user(user_id) ON DELETE CASCADE,
    FOREIGN KEY (parent_comment_id) REFERENCES comments(comment_id) ON DELETE CASCADE
) 


-- 1. 채팅방 테이블 (방의 제목이나 개설 시간 관리)
create table chat_room (
    room_id UUID not null auto_increment, -- 채팅 번호
    room_name varchar(100),      -- 채팅 이름
    created_at timestamp default now(), -- 채팅 생성일
    primary key(room_id)
)

-- 2. 채팅방 참여자 테이블 (누가 어느 방에 속해 있는지)
create table chat_participant (
    participant_id UUID not null auto_increment,  -- 참여자 번호
    room_id UUID not null,  -- 방 번호
    user_id UUID not null,  -- 유저 번호
    joined_at timestamp default now(),  -- 참가일
    primary key(participant_id),
    foreign key(room_id) references chat_room(room_id) on delete cascade,
    foreign key(user_id) references user(user_id) on delete cascade
)

-- 3. 채팅 메시지 테이블 (실제 대화 내용 저장)
create table chat_message (
    message_id UUID not null auto_increment, -- 메세지 번호
    room_id UUID not null,  -- 채팅 번호
    user_id UUID not null,  -- 유저 번호
    room_content text not null,      -- 채팅 내용      
    send_at timestamp default now(), -- 채팅 보낸날짜  
    is_read tinyint(1) default 0,      -- 읽음 표시
    primary key(message_id),
    index idx_room_send (room_id, send_at), 
    foreign key(room_id) references chat_room(room_id) on delete cascade,
    foreign key(user_id) references user(user_id) on delete cascade
)

create table group_purchase (
    gp_id UUID not null auto_increment,  -- 공동구매 번호
    user_id UUID not null,                 -- 유저 번호
    gp_title varchar(100) not null,  -- 공동구매 제목
    item_name varchar(100) not null,    -- 물품 이름
    item_price int not null,              -- 물품 가격
    target_count int not null,            -- 물품 총 인원
    current_count int default 1,          -- 현재 참여 인원
    start_date timestamp default now(), -- 시작 일
    end_date timestamp not null,          -- 모집 마감일
    status enum('OPEN', 'SUCCESS', 'FAIL', 'CANCEL') default 'OPEN', -- 물품 상태
    primary key(gp_id),
    foreign key(user_id) references user(user_id)
)

create table group_order (
    order_id varchar(50) not null,        -- 고유 주문번호
    gp_id UUID not null,                   -- 공동구매 번호
    user_id UUID not null,                 -- 유저 번호
    order_count int default 1,            -- 주문 수량
    total_price int not null,             -- 결제 금액
    
    -- 결제 통합 관리 (Toss & Kakao 공통)
    pg_provider enum('TOSS', 'KAKAO') not null, -- 결제 대행사 (TOSS 또는 KAKAO)
    pg_tid varchar(200),                  -- PG사 발급 거래 고유 키 (Toss: paymentKey / Kakao: tid)
    payment_method varchar(50),           -- 결제 수단 (CARD, MONEY, TRANSFER 등)
    
    payment_status enum('READY', 'PAID', 'CANCELLED', 'FAILED') default 'READY', -- 결제 상태
    paid_at timestamp,                    -- 결제 완료 일시
    
    primary key(order_id),
    foreign key(gp_id) references group_purchase(gp_id),
    foreign key(user_id) references user(user_id)
)
